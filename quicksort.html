// Program 7.11: quicksort C implementation
void quicksort(int list[], int left, int right) {
int pivot, i, j;
if (left < right) {
i = left;
j = right + 1;
pivot = list[left]; // 選擇最左邊為樞紐
do {
do {
i++;
} while (list[i] < pivot); // 找第一個 >= pivot
do {
j--;
} while (list[j] > pivot); // 找第一個 <= pivot
if (i < j)
SWAP(list[i], list[j]); // 交換
} while (i < j);
SWAP(list[left], list[j]); // 樞紐歸位
quicksort(list, left, j - 1); // 遞迴左側
quicksort(list, j + 1, right); // 遞迴右側
}
}
